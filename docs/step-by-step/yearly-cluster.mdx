## Methods

### divide
Divides each field by the corresponding value in another YearlyCluster instance.

```typescript
public divide(cluster: YearlyCluster): YearlyCluster {
    Object.keys(this).forEach(key => {
        if (isNaN(this[key])) {
            return;
        } else if (Object.keys(cluster).includes(key) && cluster[key] !== 0) {
            this[key] = this[key] / cluster[key];
        }
    });
    return this;
}
```
### substract
Subtracts each field by the corresponding value in another YearlyCluster instance.

```typescript
public substract(cluster: YearlyCluster): YearlyCluster {
    Object.keys(this).forEach(key => {
        if (isNaN(this[key])) {
            return;
        } else if (Object.keys(cluster).includes(key) && cluster[key] !== 0) {
            this[key] = this[key] - cluster[key];
        }
    });
    return this;
}
```
